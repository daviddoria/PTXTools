cmake_minimum_required(VERSION 2.6)

PROJECT(itkPTXReader)

FIND_PACKAGE(ITK REQUIRED)
INCLUDE(${ITK_USE_FILE})

FIND_PACKAGE(VTK REQUIRED)
INCLUDE(${VTK_USE_FILE})

ADD_EXECUTABLE(PTXtoRGB PTXtoRGB.cxx)
TARGET_LINK_LIBRARIES(PTXtoRGB ITKCommon ITKBasicFilters ITKIO
vtkHybrid vtkIO)

ADD_EXECUTABLE(PTXtoPointCloud PTXtoPointCloud.cxx)
TARGET_LINK_LIBRARIES(PTXtoPointCloud
vtkHybrid vtkIO
ITKCommon ITKBasicFilters ITKIO)
